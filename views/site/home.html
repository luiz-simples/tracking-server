Sample to send contacts

var showErrorMessage = function(messageError) {
  console.log(messageError);
};

var showSuccessMessage = function() {
  alert('Success contact send');
};

var obj = {
  name: 'Contact Full Name',
  email: 'contact@email.com',
  message: 'This is a main message'
};

trackingAnalytics.sendContact(obj.name, obj.email, obj.message, function(messageError, result) {
  if (messageError) return showErrorMessage(messageError);
  showSuccessMessage();
});
